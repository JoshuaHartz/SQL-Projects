CREATE DATABASE imdb_movies;

use imdb_movies;

CREATE TABLE directors (id INT(11), first_name VARCHAR(100), last_name VARCHAR(100), PRIMARY KEY (id));

CREATE TABLE movies (id INT(11), name VARCHAR(100), year INT(11), `rank` FLOAT, PRIMARY KEY (id));

CREATE TABLE movies_directors (director_id INT(11), movie_id INT(11), PRIMARY KEY (director_id , movie_id), FOREIGN KEY (director_id) REFERENCES directors (id), FOREIGN KEY (movie_id) REFERENCES movies (id));

CREATE TABLE directors_genres (director_id INT(11), genre VARCHAR(100), prob FLOAT, PRIMARY KEY (director_id , genre), FOREIGN KEY (director_id) REFERENCES directors (id));

CREATE TABLE movies_genres (movie_id INT(11), genre VARCHAR(100), PRIMARY KEY (movie_id, genre), FOREIGN KEY (movie_id) REFERENCES movies (id));

CREATE TABLE actors (id INT(11), first_name VARCHAR(100), last_name VARCHAR(100), gender CHAR(1), PRIMARY KEY (id));

CREATE TABLE roles (actor_id INT(11), movie_id INT(11), role VARCHAR(100),PRIMARY KEY (movie_id , actor_id, role), FOREIGN KEY (movie_id) REFERENCES movies (id), FOREIGN KEY (actor_id) REFERENCES actors (id));
